1. Fetches each student and attaches matching department details using $lookup.
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }}
])


2. Joins students with departments and displays only the student’s name and their department name. 
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$unwind: '$dept'},
  {$project: { _id: 0, name: 1, department: '$dept.dept_name'}}
])

3. Retrieves every student along with their corresponding fee information. 
db.Student.aggregate([
  {$lookup: {
    from: 'Fees',
    localField: '_id',
    foreignField: 'student_id',
    as: 'fees'
  }}
])

4. Finds students whose fee status is marked as “Unpaid”. 
db.Student.aggregate([
  {$lookup: {
    from: 'Fees',
    localField: '_id',
    foreignField: 'student_id',
    as: 'fees'
  }},
  {$unwind: '$fees'},
  {$match: {'fees.status': 'Unpaid'}}
])

5. Fetches students with both their department details and fee records in a single pipeline. 
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$lookup: {
    from: 'Fees',
    localField: '_id',
    foreignField: 'student_id',
    as: 'fees'
  }},
])

6. Returns only those students for whom no matching department exists. 
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$match: {dept: {$size: 0}}}
])

7. Selects students whose department name is "Computer Science" using join + match. 
db.Student.aggregate([

    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$unwind: '$dept'},

])

8. Performs a join with departments and sorts all students by their marks in descending order. 
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$unwind: '$dept'},
  {$project: {_id: 0, name: 1, marks: 1}},
  {$sort: {marks: -1}}
])

9. Counts how many students belong to each department after joining.
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$unwind: '$dept'},
  {$group: {
    _id: '$dept.dept_name',
    stuCount: {$sum: 1}
  }}
])

10. Joins with departments and filters students whose marks are greater than 80. 
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$unwind: '$dept'},
  {$match: {marks: {$gt: 80}}},
  {$project: {_id: 0, name: 1, marks: 1}},
  {$sort: {marks: -1}}
])

11. Finds the maximum marks obtained within each department using join + group. 
db.Student.aggregate([
  {$lookup: {
    from: 'Department',
    localField: 'dept_id',
    foreignField: '_id',
    as: 'dept'
  }},
  {$unwind: '$dept'},
  {$group: {
    _id: '$dept.dept_name',
    maxMarks: {$max: '$marks'}
  }}
])
